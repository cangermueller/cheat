d.popitem() // removed last inserted item

# namedtuple
* Like tuple p = (1, 2) but enables accessing fields by names
* Values can not be modified (like tuple)
* Similar to constant struct

Params = namedtuple("Params", ["a", "b"])
p =  Params(1, 2)
print(p.a)
print(p.b)
