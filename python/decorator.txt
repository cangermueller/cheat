# Function decorator without args
* `decorator` is called when function declaration is created, e.g. a = fun

def mult(fun):
  @functools.wraps(func) // returned function has same name as fun
  def decorator(*args, **kwargs):
    return fun(*args, **kwargs) * 2
  return decorator


# Function decorator with args
* two nested functions requires
* mult(factor) returns function _get_decorator that returns the decorater function
* @mult(factor) calls the decorater function _get_decorator that returns the decorated function

def mult(factor):
  def _get_decorator(fun):
    def decorator(*args, **kwargs):
      return fun(*args, **kwargs) * factor
    return decorator
  return _get_decorator


# Class decorator with args

def register(name):
  """Decorator for registering models."""

  def wrap(model_class):
    if name in _MODELS:
      raise ValueError('Duplicate model name "%s"!' % name)
    _MODELS[name] = model_class
    return model_class

  return wrap


# Example

def add(value=1):

  def decorator(func):

    @functools.wraps(func)
    def _new_func(*args, **kwargs):
      return func(*args, **kwargs) + value

    return _new_func

  return decorator


@add(value=2)
def mysum(a, b):
  return a + b

print(mysum(1, 2))
