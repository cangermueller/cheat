# absllog log level
* higher -> more verbose
FATAL: -3
ERROR: -2
WARN: -1
INFO: 0
DEBUG: 1

# Default logging levels
* docs.python.org/3/library/logging.html#logging-levels
NOTSET: 0
DEBUG: 10
INFO: 20
WARNING: 30 // WARN (logging.warn) is deprecated
ERROR: 40
FATAL: 50

# VLOG
* higher -> more verbose
--v=<level>  // log if level <= <level>
--vmodule=<basename>=<level>
--vmodule=gpu_compiler=1
  * file extension not needed (not gpu_compile.cc)
  * can be full path (vmodule=../service/gpu/gpu_compiler)
VLOG(N) <<
if (VLOG_IS_ON(N)) {}


# Log and exit
logging.error('Error; does not shut down')
logging.exception('Error; show exception stack trace; does not shut down')
logging.fatal('Error; shutting down with ugly stack trace')
